22void spawn(char **program) {
23 if (fork() == 0) { //fork and continue if the program is the child
24 // Child process
25 setsid(); // Start a new session (detach from controlling terminal)
26 execvp((char*)program[0], (char**)program); // Replace process image with program
27 _exit(0); //Exit once the process is over
28 }
29 // Parent process does nothing; child runs the program
30}
