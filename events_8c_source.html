42void handleEvent(xcb_generic_event_t *ev){
43 //get response type and get rid of extra bit
44 uint8_t type = ev->response_type & ~0x80;
45 //check that the handler function exists and make sure event type is within bounds for event handler array, prevent evil memory leak demons
46 if(event_handlers[type] && type < HANDLER_COUNT) {
47 event_handlers[type](ev);
48 }
49
50 else{
51 logError("Unknown event type: %u\n", 1, ev->response_type);
52 }
53}
