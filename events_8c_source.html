57void handleEvent(xcb_generic_event_t *ev){
58 //get response type and get rid of extra bit
59 uint8_t type = ev->response_type & ~0x80;
60 //check that the handler function exists and make sure event type is within bounds for event handler array, prevent evil memory leak demons
61 if(event_handlers[type] && type < HANDLER_COUNT) {
62 event_handlers[type](ev);
63 }
64
65 else{
66 logError("Unknown event type: %u\n", 1, ev->response_type);
67 }
68}

125static void handleButtonPress(xcb_generic_event_t *ev){
126 xcb_button_press_event_t *e = (xcb_button_press_event_t *) ev;
127 // redundant, but would cause weird behavior if not synced
128 foc_win = e->child;
129 focusInput(foc_win);
130 raiseWin(e->child);
131 last_button_pressed = e->detail;
132 // e-> event_x and e_event_y for some reason don't give relative coords right
133 // so we have to do them ourselves
134 // this gets use the window geometry
135 xcb_get_geometry_cookie_t win_geometry_temp = xcb_get_geometry(dpy, e->child);
136 xcb_get_geometry_reply_t *win_geometry = xcb_get_geometry_reply(dpy, win_geometry_temp, NULL);
137 // note: root_x and root_y represent the relative curser position
138 if(win_geometry){
139 window_offset.x = e->root_x - win_geometry->x;
140 window_offset.y = e->root_y - win_geometry->y;
141 // cursor needs to be moved to corner on right click
142 if((last_button_pressed == (xcb_button_t)(3)) && (foc_win != 0)){
143 xcb_warp_pointer(dpy, XCB_NONE, foc_win, 0, 0, 0, 0,
144 win_geometry->width, win_geometry->height);
145 }
146 }
147 // set cursor
148 // left mouse button
149 if(e->child){
150 if(last_button_pressed == (xcb_button_t)(1)){
151 cursor = xcb_cursor_load_cursor(cursor_context, "fleur");
152 uint32_t attribute_buf[] = {cursor};
153 xcb_change_window_attributes(dpy, scre->root, XCB_CW_CURSOR, attribute_buf);
154 }
155 // right mouse button
156 else if(last_button_pressed == (xcb_button_t)(3)){
157 cursor = xcb_cursor_load_cursor(cursor_context, "sizing");
158 uint32_t attribute_buf[] = {cursor};
159 xcb_change_window_attributes(dpy, scre->root, XCB_CW_CURSOR, attribute_buf);
160 }
161 }
162 // pointer needs to be grabbed for window movement
163 // event masks are needed so we can get passed the events
164 // cursor is passed in to force cursor icon
165 xcb_grab_pointer(dpy,0, scre->root, XCB_EVENT_MASK_BUTTON_RELEASE
166 | XCB_EVENT_MASK_BUTTON_MOTION | XCB_EVENT_MASK_POINTER_MOTION_HINT,
167 XCB_GRAB_MODE_ASYNC, XCB_GRAB_MODE_ASYNC,
168 scre->root, cursor, XCB_CURRENT_TIME);
169 xcb_flush(dpy);
170}

198static void handleMotionNotify(xcb_generic_event_t *ev){
199 (void)ev;
200 // needed to get pointer coordinates
201 // send request for pointer
202 xcb_query_pointer_cookie_t cookie = xcb_query_pointer(dpy, scre->root);
203 // receive request for pointer
204 xcb_query_pointer_reply_t *pointer = xcb_query_pointer_reply(dpy, cookie, 0);
205 // left mouse button
206 if((last_button_pressed == (xcb_button_t)(1)) && (foc_win != 0)){
207 // new x,y coord based of off offset and curser position
208 uint32_t geometry_buf[2] = {pointer->root_x - window_offset.x, pointer->root_y - window_offset.y};
209 // update window coordinates
210 xcb_configure_window(dpy, foc_win, XCB_CONFIG_WINDOW_X
211 | XCB_CONFIG_WINDOW_Y, geometry_buf);
212 }
213 // right mouse button
214 else if((last_button_pressed == (xcb_button_t)(3)) && (foc_win != 0)){
215 // get window properties
216 xcb_get_geometry_cookie_t win_geometry_temp = xcb_get_geometry(dpy, foc_win);
217 xcb_get_geometry_reply_t *win_geometry = xcb_get_geometry_reply(dpy, win_geometry_temp, NULL);
218 // prevent negative dimension windows
219 if(pointer->root_x >= win_geometry->x && pointer->root_y >= win_geometry->y){
220 uint32_t geometry_buf[2] = {pointer->root_x - win_geometry->x, pointer->root_y - win_geometry->y};
221 xcb_configure_window(dpy, foc_win, XCB_CONFIG_WINDOW_WIDTH
222 | XCB_CONFIG_WINDOW_HEIGHT, geometry_buf);
223 }
224 }
225 xcb_flush(dpy);
226}
