51void handleEvent(xcb_generic_event_t *ev){
52 //get response type and get rid of extra bit
53 uint8_t type = ev->response_type & ~0x80;
54 //check that the handler function exists and make sure event type is within bounds for event handler array, prevent evil memory leak demons
55 if(event_handlers[type] && type < HANDLER_COUNT) {
56 event_handlers[type](ev);
57 }
58
59 else{
60 logError("Unknown event type: %u\n", 1, ev->response_type);
61 }
62}

101static void handleMapNotify(xcb_generic_event_t *ev){
102 // cast the generic event to a map notify event
103 xcb_map_notify_event_t *e = (xcb_map_notify_event_t *)ev;
104 // focus the window which notified
105 foc_win = e->window;
106 raiseWin(e->window);
107 // without changing event mask the wm won't get enter notify, focusin, or focusout events
108 uint32_t event_mask = XCB_EVENT_MASK_ENTER_WINDOW | XCB_EVENT_MASK_FOCUS_CHANGE;
109 xcb_change_window_attributes_checked(dpy, e->window, XCB_CW_EVENT_MASK, &event_mask);
110 focusInput(foc_win);
111 xcb_flush(dpy);
112}

119static void handleButtonPress(xcb_generic_event_t *ev){
120 xcb_button_press_event_t *e = (xcb_button_press_event_t *) ev;
121 // redundant, but would cause weird behavior if not synced
122 foc_win = e->child;
123 focusInput(foc_win);
124 raiseWin(e->child);
125 // pointer needs to be grabbed for window movement
126 xcb_grab_pointer(dpy,0, scre->root, XCB_EVENT_MASK_BUTTON_RELEASE
127 | XCB_EVENT_MASK_BUTTON_MOTION | XCB_EVENT_MASK_POINTER_MOTION_HINT,
128 XCB_GRAB_MODE_ASYNC, XCB_GRAB_MODE_ASYNC,
129 scre->root, XCB_NONE, XCB_CURRENT_TIME);
130 last_button_pressed = e->detail;
131 xcb_flush(dpy);
132}

158 // receive request for pointer
159 xcb_query_pointer_reply_t *pointer = xcb_query_pointer_reply(dpy, cookie, 0);
160 // left mouse button
161 if((last_button_pressed == (xcb_button_t)(1)) && (foc_win != 0)){
162 // store x, y coords
163 uint32_t geometry_buf[2] = {pointer->root_x, pointer->root_y};
164 // update window coordinates
165 xcb_configure_window(dpy, foc_win, XCB_CONFIG_WINDOW_X
166 | XCB_CONFIG_WINDOW_Y, geometry_buf);
167 }
168 xcb_flush(dpy);
